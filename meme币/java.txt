// 生成随机推文
function generateRandomTweet() {
  const isKol = Math.random() < 0.2;
  const isNewToken = Math.random() < 0.15;
  const isWhale = Math.random() < 0.08;

  let user, tweetType, token;

  if (isKol) {
    user = DATA.kols[Math.floor(Math.random() * DATA.kols.length)];
    user = { ...user, isKol: true };
    tweetType = 'kol_shill';
  } else if (isWhale) {
    user = { name: 'Whale Alert', handle: '@whale_alert', avatar: '🐋', isKol: true };
    tweetType = 'whale_alert';
  } else {
    user = DATA.normalUsers[Math.floor(Math.random() * DATA.normalUsers.length)];
    user = { ...user, isKol: false };
    const types = ['normal_mention', 'normal_mention', 'normal_mention', 'pump_fun', 'general_crypto'];
    tweetType = types[Math.floor(Math.random() * types.length)];
  }

  // 选择代币
  if (isNewToken) {
    token = DATA.newTokenTemplates[Math.floor(Math.random() * DATA.newTokenTemplates.length)];
    token = { ...token, isNew: true, mentions: Math.floor(Math.random() * 50) + 1, sentiment: Math.floor(Math.random() * 40) + 50 };
  } else {
    token = DATA.knownTokens[Math.floor(Math.random() * DATA.knownTokens.length)];
    token = { ...token, isNew: false };
  }

  const templates = DATA.tweetTemplates[tweetType];
  let text = templates[Math.floor(Math.random() * templates.length)];
  text = text.replace(/\{token\}/g, token.symbol);

  const likes = isKol ? Math.floor(Math.random() * 5000) + 500 : Math.floor(Math.random() * 200);
  const retweets = isKol ? Math.floor(Math.random() * 2000) + 100 : Math.floor(Math.random() * 50);
  const replies = Math.floor(Math.random() * 100);

  return {
    id: Date.now() + Math.random(),
    user,
    text,
    token,
    isKol: user.isKol,
    isNewToken: token.isNew,
    isWhale,
    timestamp: new Date(),
    likes,
    retweets,
    replies,
    engagement: likes + retweets * 2 + replies * 3,
  };
}

// 生成趋势历史数据
function generateSparklineData(length = 20) {
  const data = [];
  let val = Math.random() * 50 + 20;
  for (let i = 0; i < length; i++) {
    val += (Math.random() - 0.4) * 15;
    val = Math.max(5, val);
    data.push(val);
  }
  return data;
}
